<VirtualHost *:80>
    ServerName ptcalculator.test
    ServerAlias pt-calculator.test
    DocumentRoot "/var/www/devroot/ptcalculator"

    # Configurar el comportamiento del servidor web en un directorio particular
    <Directory "/var/www/devroot/ptcalculator">
        #Options: Esta directiva define las opciones que se aplican al directorio especificado.
        #Options Indexes FollowSymLinks Includes ExecCGI
        Options -Indexes +FollowSymLinks +Multiviews
        Header set Access-Control-Allow-Origin "*"
        Header set Access-Control-Allow-Credentials: "true"

        #AllowOverride: Esta directiva controla qué directivas de configuración pueden ser sobrescritas por los archivos de configuración del directorio (.htaccess). En tu configuración, AllowOverride All permite que se utilicen todos los tipos de directivas de configuración en archivos .htaccess.
        AllowOverride All

        #Require all granted: Esta directiva controla el acceso al directorio y especifica quién está autorizado a acceder. En tu caso, Require all granted permite el acceso a cualquier usuario sin restricciones. Es decir, todos los usuarios tienen permiso para acceder a este directorio.
        Require all granted
    </Directory>

    ErrorLog ${APACHE_LOG_DIR}/ptcalculator-error.log
    CustomLog ${APACHE_LOG_DIR}/ptcalculator-access.log combined
</VirtualHost>
